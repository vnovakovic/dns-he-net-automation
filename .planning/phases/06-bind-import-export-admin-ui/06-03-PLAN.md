package templates

import "github.com/vnovakov/dns-he-net-automation/internal/model"

// AccountsPage renders the accounts management page.
// htmx handles inline row removal and new-account form without full page reloads.
templ AccountsPage(accounts []model.Account, data PageData) {
    @Layout(data) {
        <div class="page-header">
            <h2>Accounts</h2>
        </div>
        <div class="card">
            <table id="accounts-table" class="data-table">
                <thead>
                    <tr>
                        <th>ID</th>
                        <th>Username</th>
                        <th>Created</th>
                        <th>Actions</th>
                    </tr>
                </thead>
                <tbody>
                    for _, acc := range accounts {
                        @AccountRow(acc)
                    }
                </tbody>
            </table>
        </div>
        <div class="card" style="margin-top:1rem;">
            <h3>Register Account</h3>
            <p class="text-muted">Account ID must match the Vault path for the HE credentials.</p>
            <form
                hx-post="/admin/accounts"
                hx-target="#accounts-table tbody"
                hx-swap="beforeend"
                hx-on::after-request="this.reset()">
                <div class="form-row">
                    <div class="form-group">
                        <label for="account_id">Account ID</label>
                        <input type="text" id="account_id" name="account_id" placeholder="e.g. primary" required/>
                    </div>
                    <div class="form-group">
                        <label for="username">HE Username</label>
                        <input type="text" id="username" name="username" placeholder="your@email.com" required/>
                    </div>
                    <button type="submit" class="btn btn-primary">Register</button>
                </div>
            </form>
        </div>
    }
}

// AccountRow renders a single account table row.
// Used both for full-page render and as htmx response for new account creation.
templ AccountRow(acc model.Account) {
    <tr id={ "account-row-" + acc.ID }>
        <td><code>{ acc.ID }</code></td>
        <td>{ acc.Username }</td>
        <td>{ acc.CreatedAt.Format("2006-01-02 15:04") }</td>
        <td>
            <button
                class="btn btn-danger btn-sm"
                hx-delete={ "/admin/accounts/" + acc.ID }
                hx-target={ "#account-row-" + acc.ID }
                hx-swap="outerHTML swap:500ms"
                hx-confirm={ "Remove account '" + acc.ID + "' and all its tokens?" }>
                Remove
            </button>
        </td>
    </tr>
}

func handleAccountCreate(db *sql.DB) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        if err := r.ParseForm(); err != nil {
            http.Error(w, "Bad request", http.StatusBadRequest)
            return

        accountID := r.FormValue("account_id")

        if err != nil {

            return

    }
}

package templates

import (
    "fmt"
    "github.com/vnovakov/dns-he-net-automation/internal/model"
    "github.com/vnovakov/dns-he-net-automation/internal/token"

// TokensPage renders the top-level tokens page: list of accounts,
// each with an "Expand" link to load tokens via htmx.
templ TokensPage(accounts []model.Account, data PageData) {
    @Layout(data) {
        <div class="page-header">
            <h2>Tokens</h2>
        </div>
        for _, acc := range accounts {
            <div class="card" style="margin-bottom:1rem;">
                <div class="card-header">
                    <strong>{ acc.ID }</strong>
                    <span class="text-muted">({ acc.Username })</span>
                    <button
                        class="btn btn-sm"
                        hx-get={ "/admin/tokens/" + acc.ID }
                        hx-target={ "#tokens-" + acc.ID }
                        hx-swap="innerHTML">
                        Load Tokens
                    </button>
                </div>
                <div id={ "tokens-" + acc.ID }>
                    <!-- Tokens loaded on demand via htmx -->
                </div>
            </div>
        }
    }
}

// TokensForAccount renders the token list for a single account.
// Used as the htmx partial response for "Load Tokens" button.
templ TokensForAccount(accountID string, tokens []token.TokenRecord) {
    <table class="data-table" style="margin-top:0.75rem;">
        <thead>
            <tr>
                <th>ID</th>
                <th>Label</th>
                <th>Role</th>
                <th>Created</th>
                <th>Expires</th>
                <th>Status</th>
                <th>Actions</th>
            </tr>
        </thead>
        <tbody id={ "token-rows-" + accountID }>
            for _, tok := range tokens {
                @TokenRow(tok)
            }
        </tbody>
    </table>
    <div style="margin-top:0.75rem;">
        @IssueTokenForm(accountID)
    </div>
}

// TokenRow renders a single token row.
templ TokenRow(tok token.TokenRecord) {
    <tr id={ "token-row-" + tok.JTI }>
        <td><code>{ tok.JTI[:8] }...</code></td>
        <td>{ labelOrDash(tok.Label) }</td>
        <td>
            if tok.Role == "admin" {
                <span class="badge badge-admin">admin</span>
            } else {
                <span class="badge">viewer</span>
            }
        </td>
        <td>{ tok.CreatedAt.Format("2006-01-02") }</td>
        <td>{ expiresStr(tok) }</td>
        <td>
            if tok.RevokedAt != nil {
                <span class="badge badge-danger">revoked</span>
            } else {
                <span class="badge badge-success">active</span>
            }
        </td>
        <td>
            if tok.RevokedAt == nil {
                <button
                    class="btn btn-danger btn-sm"
                    hx-delete={ "/admin/tokens/" + tok.JTI }
                    hx-target={ "#token-row-" + tok.JTI }
                    hx-swap="outerHTML swap:500ms"
                    hx-confirm="Revoke this token? It will be immediately rejected on the next request.">
                    Revoke
                </button>
            }
        </td>
    </tr>
}

// IssueTokenForm renders the inline form for issuing a new token.
templ IssueTokenForm(accountID string) {
    <form
        hx-post={ "/admin/tokens/" + accountID }
        hx-target={ "#token-rows-" + accountID }
        hx-swap="beforeend">
        <div class="form-row">
            <div class="form-group">
                <label>Label</label>
                <input type="text" name="label" placeholder="e.g. ansible-prod"/>
            </div>
            <div class="form-group">
                <label>Role</label>
                <select name="role">
                    <option value="viewer">viewer</option>
                    <option value="admin">admin</option>
                </select>
            </div>
            <button type="submit" class="btn btn-primary">Issue Token</button>
        </div>
    </form>
}

// NewTokenResult renders the issued token row with the plaintext token shown once.
// This is the htmx response after POST /admin/tokens/{accountID}.
// Token value is shown only here — never again. (SEC-02 pattern: displayed once at creation)
templ NewTokenResult(tok token.TokenRecord, rawToken string) {
    @TokenRow(tok)
    <tr>
        <td colspan="7" class="token-reveal">
            <strong>Token (copy now — shown once):</strong>
            <code>{ rawToken }</code>
        </td>
    </tr>
}

// Helper functions for template rendering.
func labelOrDash(label *string) string {
    if label == nil || *label == "" {
        return "—"
    }
    return *label
}

func expiresStr(tok token.TokenRecord) string {
    if tok.ExpiresAt == nil {
        return "never"
    }
    return tok.ExpiresAt.Format("2006-01-02")
}
